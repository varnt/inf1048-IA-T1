python pacman.py
# Executa o Pacman com as configurações padrão (layout e agente padrão).

python pacman.py --layout testMaze --pacman GoWestAgent
# Usa o layout "testMaze" com o agente GoWestAgent, que sempre tenta ir para o oeste (West).

python pacman.py --layout tinyMaze --pacman GoWestAgent
# Mesmo agente GoWestAgent, agora no layout "tinyMaze".

python pacman.py -h
# Exibe o menu de ajuda com todas as opções disponíveis.

python pacman.py -l tinyMaze -p SearchAgent -a fn=tinyMazeSearch
# Executa SearchAgent usando a função tinyMazeSearch, solução fixa exclusiva do tinyMaze.

python pacman.py -l tinyMaze -p SearchAgent
# Executa SearchAgent no tinyMaze usando a função de busca padrão (normalmente DFS).

python pacman.py -l mediumMaze -p SearchAgent
# Executa DFS no mediumMaze. Não garante caminho ótimo.

python pacman.py -l bigMaze -z .5 -p SearchAgent
# Executa DFS no bigMaze com zoom reduzido para 50%.

python pacman.py -l mediumMaze -p SearchAgent -a fn=bfs
# Executa BFS no mediumMaze. BFS encontra o caminho mais curto.

python pacman.py -l bigMaze -p SearchAgent -a fn=bfs -z .5
# Executa BFS no bigMaze com zoom 0.5. Busca ótima mas mais lenta.
